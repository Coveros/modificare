172: 		coverPickSets.add(test.getTest());

174: 		Iterator requirementSubsetToRemoveIterator =

281: 	 *  This method finds the test with the most coverage, shortest time, or largest coverage per cost 
282: 	 *  ratio of the given list of requirements and returns it.  
283: 	 *  It recursively solves ties.

285: 	 * @param testList:

290: 	private SingleTestSubset getBestSingleTestSubset(LinkedHashSet testList,
291: 																	LinkedHashSet requirementList,
292: 																	int cardToCheck, int looksLeft, String metric) 
293: 	{
294: 
295: 		SingleTest bestSoFar = null;
296: 		int bestSoFarCardinalityCovers;
297: 		int competitorCardinalityCovers;
298: 		SingleTestSubset bestSoFarSingleTestSubset = null;
299: 		LinkedHashSet tieList = new LinkedHashSet();
300: 				
301: 		// Extract an iterator over the covering tests
302: 		Iterator coveringTestsIterator =	testList.iterator();
303: 	
304: 		// Set the first one as the best choice so far.  In this way
305: 		// if there is only one, it will be selected.  
306: 		
307: 		bestSoFar = (SingleTest) coveringTestsIterator.next();
308: 		
309: 		// Get the SingleTestSubset for the bestSoFar SingleTest
310: 		// Note: This is a wasted operation for single cardinality
311: 		// requirementSubsets but it will be less of loss to put it
312: 		// here than to re-update it unneccesarily in every
313: 		// iteration of the next loop.
314: 		bestSoFarSingleTestSubset = 
315: 			getSingleTestSubsetFromSingleTest(bestSoFar);
316: 
317: 	   	// It will never be the case that this method will return null because we have
318: 			// rebuilt the the requirementSubsetsSet to contain only requirementSubsets
319: 			// that are covered by a remaining test. 
320: 		
321: 		if (looksLeft < 0) {
322: 			return bestSoFarSingleTestSubset;
323: 		}
324: 
325: 							
326: 		// While there are still more covering tests to be examined.
327: 		while (coveringTestsIterator.hasNext())
328: 		{
329: 			// Get the current SingleTestSubset
330: 			SingleTestSubset currentSingleTestSubsetFromRS =
331: 				getSingleTestSubsetFromSingleTest((SingleTest)coveringTestsIterator.next());
332: 				
333: 			if(metric.equals("time"))
334: 			{
335: 					
336: 				// Uses time as the greedy choice factor
337: 				if ( currentSingleTestSubsetFromRS.getTest().getCost() < bestSoFarSingleTestSubset.getTest().getCost()) 
338: 				{
339: 					bestSoFarSingleTestSubset = currentSingleTestSubsetFromRS;
340: 					tieList.clear();				
341: 				} 									
342: 						
343: 				// If it is a tie
344: 				else if (currentSingleTestSubsetFromRS.getTest().getCost() == bestSoFarSingleTestSubset.getTest().getCost())
345: 				{
346: 					if (!tieList.contains(bestSoFar)) 
347: 						tieList.add(bestSoFar);	
348: 			
349: 					tieList.add(currentSingleTestSubsetFromRS.getTest());
350: 				}
351: 			}
352: 			
353: 			else if(metric.equals("ratio"))
354: 			{
355: 			
356: 				double bestSoFarCost = bestSoFarSingleTestSubset.getTest().getCost();
357: 				double competitorCost = currentSingleTestSubsetFromRS.getTest().getCost();
358: 				
359: 				// Get the number of current cardinality requirements that
360: 				// the bestSoFar covers.
361: 				bestSoFarCardinalityCovers = 
362: 					getNumberOfCurrentCardinalityCovers(bestSoFarSingleTestSubset,
363: 																	requirementList);	
364: 			
365: 				// Get the covers
366: 				competitorCardinalityCovers =
367: 					getNumberOfCurrentCardinalityCovers(currentSingleTestSubsetFromRS,
368: 																	requirementList);
369: 			
370: 				
371: 				double bestSoFarRatio = bestSoFarCardinalityCovers/bestSoFarCost;
372: 				double competitorRatio = competitorCardinalityCovers/competitorCost;
373: 				
374: 				// Uses coverage as the greedy choice factor
375: 				if ( competitorRatio > bestSoFarRatio) 
376: 				{
377: 					bestSoFarSingleTestSubset = currentSingleTestSubsetFromRS;
378: 					bestSoFarRatio = competitorRatio;
379: 					tieList.clear();				
380: 				} 									
381: 						
382: 				// If it is a tie
383: 				else if (competitorRatio == bestSoFarRatio)
384: 				{
385: 					if (!tieList.contains(bestSoFar)) 
386: 						tieList.add(bestSoFar);	
387: 			
388: 					tieList.add(currentSingleTestSubsetFromRS.getTest());
389: 				}
390: 			}
391: 			
392: 			else if(metric.equals("coverage"))
393: 			{
394: 			
395: 				// Get the number of current cardinality requirements that
396: 				// the bestSoFar covers.
397: 				// Note:  Same as above.
398: 				bestSoFarCardinalityCovers = 
399: 					getNumberOfCurrentCardinalityCovers(bestSoFarSingleTestSubset,
400: 																	requirementList);	
401: 				// Now we need to check to see if it covers more current 
402: 				// cardinality RequirementSubsets than the bestSoFar 
403: 				// SingleTest object 
404: 				competitorCardinalityCovers =
405: 					getNumberOfCurrentCardinalityCovers(currentSingleTestSubsetFromRS,
406: 																	requirementList);
407: 			
408: 			
409: 				// Uses coverage as the greedy choice factor
410: 				if ( competitorCardinalityCovers > bestSoFarCardinalityCovers) 
411: 				{
412: 					bestSoFarSingleTestSubset = currentSingleTestSubsetFromRS;
413: 					bestSoFarCardinalityCovers = competitorCardinalityCovers;
414: 					tieList.clear();				
415: 				} 									
416: 						
417: 				// If it is a tie
418: 				else if (competitorCardinalityCovers == bestSoFarCardinalityCovers)
419: 				{
420: 					if (!tieList.contains(bestSoFar)) 
421: 						tieList.add(bestSoFar);	
422: 			
423: 					tieList.add(currentSingleTestSubsetFromRS.getTest());
424: 				}
425: 			}
426: 		}	
427: 			if (tieList.size() != 0 && !metric.equals("time")) {
428: 				return getBestSingleTestSubset(tieList,
429: 														(buildCardinalityList(cardToCheck+1)),
430: 														(cardToCheck+1),
431: 														(looksLeft-1),metric); 					
432: 			}
433: 
434: 			return bestSoFarSingleTestSubset;
435: 	}

440: 	public void reduceUsingHGS(String metric) {
441: 	
442: 	// Check to see if the metric value is legit.
443: 		if (!metric.equals("coverage") && !metric.equals("time") && !metric.equals("ratio"))
444: 		{
445: 			System.out.println("\nInvalid metric type.\nProgram terminated");  
446: 			System.exit(0);
447: 		}
448: 		
449: 		// I have no idea what this was used for.  It apparently is not necessary now.
450: 		// int[] deathArray = new int[1];
451: 		
452: 		// The list of covering tests
453: 		coverPickSets = new LinkedHashSet();
454: 		
455: 		// This is the list that holds all of the current cardinality 
456: 		// RequirementSubsets.
457: 		LinkedHashSet requirementsOfCardinality = new LinkedHashSet();		
458: 		
459: 		// Repeat the loop until all of the requirements are covered.
460: 		// i.e. There are no more RequirementSubsets in the
461: 		// requirementSubsetUniverse.
462: 	 
463: 	   while (!cover.getRequirementSubsetUniverse().isEmpty()) {
464: 			// Update the cardinality
465: 	      cardinality++;
466: 			
467: 			// Populate the list of requirements that are of the current cardinality.
468: 			requirementsOfCardinality = buildCardinalityList(cardinality);
469: 		
470: 			// While the requirements of cardinality are not all covered
471: 			while (!requirementsOfCardinality.isEmpty()){
472: 			
473: 				// Extract an iterator of the the requirements of cardinality.
474: 				Iterator requirementsOfCardinalityIterator = 
475: 					requirementsOfCardinality.iterator();
476: 			
477: 				// Get the next requirement of cardinality.
478: 				RequirementSubset currentRequirementSubset = 
479: 					(RequirementSubset) requirementsOfCardinalityIterator.next();
480: 
481: 				// Get the list of tests that cover the current requirement 
482: 				LinkedHashSet coveringTests = currentRequirementSubset.getCoveringTests();
483: 
484: 				addTestToReducedSuite(getBestSingleTestSubset(coveringTests,
485: 																				 requirementsOfCardinality,
486: 																				 cardinality, 
487: 																				 numberOfLooksAhead,metric),
488: 																				 requirementsOfCardinality);
489: 			}
490: 		}

494: 	public void prioritizeUsingHGS(String metric) {
495: 		//This is the set that will be added to the prioritized cover
496: 		prioritizedSets = new LinkedHashSet();
497: 		LinkedHashSet remainingTests;			
498: 		    
499: 		LinkedHashSet requirementsBeforeReduction = new LinkedHashSet(); 
500: 
501: 		requirementsBeforeReduction.addAll((LinkedHashSet) cover.getRequirementSubsetUniverse());
502: 
503: 		// keep track of how many times the reduction algorithm is completed.
504: 			int i = 0;		 	
505: 		
506: 		// While there are still tests left
507: 		while (!cover.getTestSubsets().isEmpty()) {
508: 			remainingTests = new LinkedHashSet();			
509: 			
510: 			// Reset the cardinality each time.
511: 			cardinality = 0;
512: 		
513: 			// Reduce the current SetCover
514: 			this.reduceUsingHGS(metric);
515: 
516: 			// Update the number of times the reduction algorithm has been run.
517: 			i++;	
518: 			
519: 			// Reset the requirementSubsetUniverse
520: 			this.cover.addRequirementSubsets(requirementsBeforeReduction);
521: 			
522: 			// Add the reduction results to the prioritizedSets list
523: 			this.prioritizedSets.addAll(coverPickSets);		
524: 
525: 			// Extract an iterator over the testSubsets
526: 			Iterator testSubsetsIterator = this.cover.getTestSubsets().iterator();
527: 
528: 			// This loop constructs the list of remainingTests
529: 			while (testSubsetsIterator.hasNext()){
530: 				// add the test to the remainingTests list
531: 				remainingTests.add(((SingleTestSubset) testSubsetsIterator.next()).getTest());
532: 			}
533: 			
534: 						
535:     		//Extract an iterator over the requirementSubsetUniverse
536: 			Iterator requirementSubsetIterator = cover.getRequirementSubsetUniverse().iterator();
537: 
538: 			LinkedHashSet requirementsToRemove = new LinkedHashSet();
539: 
540: 			// This loop removes the covering tests from each requirement that are no longer 
541: 			// in the remainingTests list.  If this removes all of the tests for a RS, then
542: 			// the RS is added to the list to be removed.
543: 			while (requirementSubsetIterator.hasNext()) {
544: 				// Get the current RequirementSubset
545: 				RequirementSubset currentRS = (RequirementSubset) requirementSubsetIterator.next();
546: 			
547: 				LinkedHashSet currentCTS = currentRS.getCoveringTests();
548: 
549: 				currentCTS.retainAll(remainingTests);
550: 
551: 				if (currentCTS.size() == 0) {
552: 					requirementsToRemove.add(currentRS);
553: 				}
554: 			}
555: 			
556: 			cover.removeRequirementSubsets(requirementsToRemove);
557: 		}
558: 	}
